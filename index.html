function getWeightedIndex(names, selections) { const weightedPool = []; for (let i = 0; i < names.length; i++) { const name = names[i]; const selectionCount = selections[i]; if (selectionCount === 0) { // 3x odds for names with 0 selections weightedPool.push(name, name, name); } else if (selectionCount <= 5) { // 2x odds for names in the bottom 5 selections weightedPool.push(name, name); } else { // 1x odds for all other names weightedPool.push(name); } } const randomIndex = Math.floor(Math.random() * weightedPool.length); return weightedPool[randomIndex]; } // Example usage: const names = ['Alice', 'Bob', 'Charlie', 'David', 'Eve']; const selections = [0, 1, 2, 3, 4]; // example selection counts const selectedName = getWeightedIndex(names, selections); console.log(selectedName);
